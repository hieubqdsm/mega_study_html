Stat System

45 phút

Đây là xương sống của mọi RPG. Làm đúng từ đầu — mọi thứ khác (combat, loot, level up) đều phụ thuộc vào đây.

◆ Stat cơ bản (đủ cho Hack & Slash)
STRENGTH+Physical DMGdmg = str × 2.5 + weapon
INTELLIGENCE+Spell DMGspell = int × 3 + staff
DEXTERITY+Crit / Dodgecrit% = dex × 0.3
VITALITY+Max HPhp = 100 + vit × 15
DEFENSE-Dmg takenreduction = def / (def + 100)
MOVE SPEEDpx/sbase 200 + boots bonus

StatComponent.gd — Tái sử dụng cho mọi entity

# stat_component.gd — gắn vào Player VÀ Enemy
extends Node
class_name StatComponent

# Base stats (từ class/race)
@export var base_strength:     int = 10
@export var base_intelligence: int = 5
@export var base_dexterity:    int = 8
@export var base_vitality:     int = 10
@export var base_defense:      int = 5

# Bonus từ equipment (cộng dồn khi equip)
var bonus_strength:     int = 0
var bonus_intelligence: int = 0
var bonus_dexterity:    int = 0
var bonus_vitality:     int = 0
var bonus_defense:      int = 0
var bonus_move_speed:   float = 0.0

# ── Computed properties ──
var strength:     int: get: return base_strength     + bonus_strength
var intelligence: int: get: return base_intelligence + bonus_intelligence
var dexterity:    int: get: return base_dexterity    + bonus_dexterity
var vitality:     int: get: return base_vitality     + bonus_vitality
var defense:      int: get: return base_defense      + bonus_defense

# ── Derived stats ──
var max_hp:       int:   get: return 100 + vitality * 15
var phys_damage:  int:   get: return strength * 2
var spell_damage: int:   get: return intelligence * 3
var crit_chance:  float: get: return dexterity * 0.003   # 0-1
var dodge_chance: float: get: return dexterity * 0.002
var move_speed:   float: get: return 200.0 + bonus_move_speed

# Defense formula (Diablo-style: diminishing returns)
func get_damage_reduction() -> float:
    return float(defense) / (float(defense) + 100.0)   # max ~0.9

# Apply bonus khi equip/unequip
func apply_item_bonus(item: ItemData, equipping: bool):
    var sign = 1 if equipping else -1
    bonus_strength     += item.bonus_str   * sign
    bonus_intelligence += item.bonus_int   * sign
    bonus_dexterity    += item.bonus_dex   * sign
    bonus_vitality     += item.bonus_vit   * sign
    bonus_defense      += item.bonus_def   * sign
    bonus_move_speed   += item.bonus_speed * sign
◆ Dùng Resource cho ItemData Tạo class_name ItemData extends Resource với các @export var. Lưu thành file .tres. Dễ tạo trong editor, dễ serialize, dễ thêm item mới mà không cần code.

player.gd — Movement & Attack

extends CharacterBody2D

@onready var stats     = $StatComponent
@onready var health    = $HealthComponent
@onready var skills    = $SkillComponent
@onready var sprite    = $AnimatedSprite2D
@onready var wpn_pivot = $WeaponPivot

var facing_dir: Vector2 = Vector2.RIGHT
var is_attacking: bool = false

func _ready():
    health.died.connect(_on_died)
    add_to_group("player")

func _physics_process(delta):
    _move(delta)
    _rotate_weapon_to_mouse()
    _update_animation()
    move_and_slide()

func _unhandled_input(event):
    # Left click = basic attack
    if event is InputEventMouseButton:
        if event.button_index == MOUSE_BUTTON_LEFT and event.pressed:
            skills.use_skill("basic_attack", get_global_mouse_position())

    # Skill hotkeys
    if event is InputEventKey and event.pressed:
        match event.keycode:
            KEY_Q: skills.use_skill("skill_1", get_global_mouse_position())
            KEY_W: skills.use_skill("skill_2", get_global_mouse_position())
            KEY_E: skills.use_skill("skill_3", get_global_mouse_position())
            KEY_R: skills.use_skill("ultimate", get_global_mouse_position())
            KEY_SPACE: _dash()

# ===== MOVEMENT (iso 8-dir) =====
func _move(_delta):
    if is_attacking: return   # lock movement khi attack (tùy chọn)
    var input = Vector2(
        Input.get_axis("move_left", "move_right"),
        Input.get_axis("move_up",   "move_down")
    )
    if input.length() > 0:
        facing_dir = input.normalized()
        # Iso conversion
        velocity = Vector2(
            input.x - input.y,
            (input.x + input.y) * 0.5
        ).normalized() * stats.move_speed
    else:
        velocity = velocity.lerp(Vector2.ZERO, 0.2)

# ===== WEAPON ROTATION =====
func _rotate_weapon_to_mouse():
    var mouse = get_global_mouse_position()
    var angle = (mouse - global_position).angle()
    wpn_pivot.rotation = angle
    # Flip sprite theo hướng chuột
    sprite.flip_h = mouse.x < global_position.x

# ===== DASH =====
var dash_cooldown: bool = false
func _dash():
    if dash_cooldown: return
    dash_cooldown = true
    velocity = facing_dir * 800.0
    # Invincible during dash
    health.invincible = true
    await get_tree().create_timer(0.2).timeout
    health.invincible = false
    await get_tree().create_timer(0.8).timeout  # cooldown
    dash_cooldown = false

func _on_died():
    get_tree().change_scene_to_file("res://scenes/ui/GameOver.tscn")

HealthComponent.gd

# health_component.gd — dùng cho Player VÀ Enemy
extends Node
class_name HealthComponent

signal health_changed(current: int, maximum: int)
signal died

var current_hp: int
var invincible: bool = false

@onready var stats: StatComponent = get_parent().get_node("StatComponent")

func _ready():
    current_hp = stats.max_hp

func take_damage(raw_damage: int, damage_type: String = "physical") -> int:
    if invincible: return 0

    # Dodge check
    if randf() < stats.dodge_chance:
        EventBus.show_damage_number.emit(0, get_parent().global_position, "dodge")
        return 0

    # Defense reduction
    var final_dmg = raw_damage
    if damage_type == "physical":
        final_dmg = max(1, int(raw_damage * (1.0 - stats.get_damage_reduction())))

    # Critical hit (attacker's crit, not defender's)
    # (passed in raw_damage already, see SkillComponent)

    current_hp = max(0, current_hp - final_dmg)
    health_changed.emit(current_hp, stats.max_hp)
    EventBus.show_damage_number.emit(final_dmg, get_parent().global_position, damage_type)
    EventBus.entity_damaged.emit(get_parent(), final_dmg, damage_type)

    if current_hp <= 0:
        died.emit()
        EventBus.entity_died.emit(get_parent(), get_parent().global_position)

    return final_dmg

func heal(amount: int):
    current_hp = min(current_hp + amount, stats.max_hp)
    health_changed.emit(current_hp, stats.max_hp)

SkillComponent.gd

# skill_component.gd
extends Node
class_name SkillComponent

@export var skills: Array[SkillData] = []

var cooldowns: Dictionary = {}     # skill_id → time remaining
var current_mana: int = 100
var max_mana: int = 100

@onready var stats: StatComponent = get_parent().get_node("StatComponent")

func _process(delta):
    # Giảm cooldown mỗi frame
    for id in cooldowns.keys():
        cooldowns[id] = max(0, cooldowns[id] - delta)
    # Regen mana
    current_mana = min(max_mana, current_mana + 5 * delta)

func use_skill(skill_id: String, target_pos: Vector2) -> bool:
    var skill = _get_skill(skill_id)
    if not skill: return false
    if cooldowns.get(skill_id, 0.0) > 0: return false
    if current_mana < skill.mana_cost: return false

    # Pay cost
    current_mana -= skill.mana_cost
    cooldowns[skill_id] = skill.cooldown

    # Execute
    if skill.projectile:
        _fire_projectile(skill, target_pos)
    elif skill.aoe_radius > 0:
        _execute_aoe(skill, target_pos)
    else:
        _execute_melee(skill, target_pos)

    return true

func _calculate_damage(skill: SkillData) -> int:
    var base = stats.phys_damage if skill.damage_type == "physical" else stats.spell_damage
    var dmg = int(base * skill.damage_mult)
    # Crit check
    if randf() < stats.crit_chance:
        dmg = int(dmg * 2.0)   # 200% crit
        # Tag damage as crit for red number display
    return dmg

func _execute_melee(skill: SkillData, target_pos: Vector2):
    var dmg = _calculate_damage(skill)
    var dir = (get_parent().global_position - target_pos).normalized()
    # Tìm enemy trong range theo hướng
    var enemies = get_tree().get_nodes_in_group("enemies")
    for e in enemies:
        if e.global_position.distance_to(get_parent().global_position) <= skill.range:
            e.get_node("HealthComponent").take_damage(dmg, skill.damage_type)

func _execute_aoe(skill: SkillData, center: Vector2):
    var dmg = _calculate_damage(skill)
    for e in get_tree().get_nodes_in_group("enemies"):
        if e.global_position.distance_to(center) <= skill.aoe_radius:
            e.get_node("HealthComponent").take_damage(dmg, skill.damage_type)

func _fire_projectile(skill: SkillData, target_pos: Vector2):
    var proj = skill.projectile.instantiate()
    get_tree().root.add_child(proj)
    proj.global_position = get_parent().global_position
    proj.setup(target_pos, _calculate_damage(skill), skill.damage_type, skill.aoe_radius)

func _get_skill(id: String) -> SkillData:
    for s in skills:
        if s.id == id: return s
    return null

Projectile.gd — Fireball / Arrow

# projectile.gd
extends Area2D

var damage:       int
var damage_type:  String
var aoe_radius:   float
var speed:        float = 400.0
var direction:    Vector2
var lifetime:     float = 3.0

func setup(target: Vector2, dmg: int, dtype: String, aoe: float):
    damage      = dmg
    damage_type = dtype
    aoe_radius  = aoe
    direction   = (target - global_position).normalized()
    rotation    = direction.angle()

func _ready():
    body_entered.connect(_on_hit)

func _physics_process(delta):
    position += direction * speed * delta
    lifetime -= delta
    if lifetime <= 0: queue_free()

func _on_hit(body):
    if body.is_in_group("player"): return   # không tự bắn mình

    if aoe_radius > 0:
        # AOE explosion
        for e in get_tree().get_nodes_in_group("enemies"):
            if e.global_position.distance_to(global_position) <= aoe_radius:
                e.get_node("HealthComponent").take_damage(damage, damage_type)
        # Spawn explosion VFX
        EventBus.item_dropped.emit(null, global_position)   # reuse for VFX signal
    else:
        if body.has_node("HealthComponent"):
            body.get_node("HealthComponent").take_damage(damage, damage_type)

    queue_free()

# enemy.gd
extends CharacterBody2D

enum State { IDLE, PATROL, CHASE, ATTACK, STAGGERED, DEAD }

@export var enemy_data: EnemyData   # Resource chứa stats, loot table...
var state: State = State.IDLE
var player: Node2D
var nav_path: Array[Vector2] = []

@onready var stats  = $StatComponent
@onready var health = $HealthComponent
@onready var sprite = $AnimatedSprite2D
@onready var nav_timer = $NavTimer   # wait 0.3s
@onready var attack_timer = $AttackTimer

const DETECT_RANGE = 250.0
const ATTACK_RANGE = 70.0
const LOSE_RANGE   = 450.0

func _ready():
    add_to_group("enemies")
    player = get_tree().get_first_node_in_group("player")
    health.died.connect(_on_died)
    nav_timer.timeout.connect(_recalc_nav)

func _physics_process(delta):
    match state:
        State.IDLE:    _state_idle()
        State.CHASE:   _state_chase(delta)
        State.ATTACK:  _state_attack()
        State.STAGGERED: pass
    move_and_slide()

# ── IDLE ──
func _state_idle():
    velocity = Vector2.ZERO
    if player and _dist_to_player() < DETECT_RANGE:
        state = State.CHASE
        nav_timer.start()

# ── CHASE (dùng pathfinding) ──
func _state_chase(delta):
    if _dist_to_player() > LOSE_RANGE:
        state = State.IDLE; return
    if _dist_to_player() < ATTACK_RANGE:
        state = State.ATTACK; return

    if nav_path.size() > 0:
        var target = nav_path[0]
        var dir = (target - global_position).normalized()
        velocity = dir * stats.move_speed
        sprite.flip_h = velocity.x < 0
        sprite.play("walk")
        if global_position.distance_to(target) < 10:
            nav_path.remove_at(0)

# ── ATTACK ──
func _state_attack():
    velocity = Vector2.ZERO
    if _dist_to_player() > ATTACK_RANGE * 1.5:
        state = State.CHASE; return

    if attack_timer.is_stopped():
        sprite.play("attack")
        var dmg = int(stats.phys_damage * enemy_data.attack_mult)
        player.get_node("HealthComponent").take_damage(dmg)
        attack_timer.start(enemy_data.attack_speed)

# ── DIE ──
func _on_died():
    state = State.DEAD
    # Drop XP + Loot
    EventBus.xp_gained.emit(enemy_data.xp_reward)
    LootSystem.roll_loot(enemy_data.loot_table, global_position)
    sprite.play("die")
    await sprite.animation_finished
    queue_free()

# ── STAGGER khi bị hit ──
func stagger(duration: float = 0.3):
    var prev = state
    state = State.STAGGERED
    sprite.play("hurt")
    await get_tree().create_timer(duration).timeout
    state = prev

func _dist_to_player() -> float:
    return global_position.distance_to(player.global_position) if player else 9999

func _recalc_nav():
    if player and state == State.CHASE:
        nav_path = get_node("/root/Main/Pathfinder").find_path(global_position, player.global_position)

Level Up System

40 phút 
# xp_component.gd — gắn vào Player
extends Node
class_name XPComponent

var level:       int = 1
var current_xp:  int = 0
var stat_points: int = 0   # phân bổ khi lên level
var skill_points:int = 0

@onready var stats: StatComponent = get_parent().get_node("StatComponent")

# XP cần để lên level: 100 × level^1.5 (Diablo-style curve)
func xp_needed() -> int:
    return int(100 * pow(level, 1.5))

func _ready():
    EventBus.xp_gained.connect(gain_xp)

func gain_xp(amount: int):
    current_xp += amount
    while current_xp >= xp_needed():
        current_xp -= xp_needed()
        _level_up()

func _level_up():
    level += 1
    stat_points  += 3    # phân bổ vào STR/INT/DEX/VIT/DEF
    skill_points += 1    # mở skill mới

    # Auto-increase base stats nhỏ
    stats.base_strength     += 1
    stats.base_vitality     += 1

    EventBus.level_up.emit(level, stat_points)
    # Heal lên full khi lên level (Diablo 1 style)
    get_parent().get_node("HealthComponent").heal(9999)

# ── PHÂN BỔ STAT ──
func invest_stat(stat_name: String) -> bool:
    if stat_points <= 0: return false
    stat_points -= 1
    match stat_name:
        "strength":     stats.base_strength     += 2
        "intelligence": stats.base_intelligence += 2
        "dexterity":    stats.base_dexterity    += 2
        "vitality":     stats.base_vitality     += 2
        "defense":      stats.base_defense      += 2
    return true

# loot_system.gd (Autoload)
extends Node

@export var item_drop_scene: PackedScene

# Rarity weights (dựa theo player MagicFind stat)
func roll_loot(loot_table: Array[Dictionary], spawn_pos: Vector2):
    for entry in loot_table:
        # entry = { item_id, weight, min_qty, max_qty }
        if randf() <= entry["weight"]:
            var qty = randi_range(entry["min_qty"], entry["max_qty"])
            var item = ItemDB.get_item(entry["item_id"])
            for i in qty:
                _spawn_drop(item, spawn_pos + Vector2(
                    randf_range(-30, 30),
                    randf_range(-20, 20)
                ))

func _spawn_drop(item: ItemData, pos: Vector2):
    var drop = item_drop_scene.instantiate()
    get_tree().root.add_child(drop)
    drop.global_position = pos
    drop.setup(item)

# inventory.gd — gắn vào Player
extends Node
class_name InventoryComponent

const MAX_SLOTS = 30

# Grid inventory [slot_idx] → ItemData | null
var slots: Array = []

# Equipment slots
var equipped: Dictionary = {
    "weapon": null,
    "helmet": null,
    "chest":  null,
    "boots":  null,
    "ring_1": null,
    "ring_2": null,
}

signal inventory_changed
signal item_equipped(slot, item)

@onready var stats: StatComponent = get_parent().get_node("StatComponent")

func _ready():
    slots.resize(MAX_SLOTS)
    slots.fill(null)
    EventBus.item_picked_up.connect(add_item)

func add_item(item: ItemData) -> bool:
    var empty_slot = slots.find(null)
    if empty_slot == -1: return false   # đầy
    slots[empty_slot] = item
    inventory_changed.emit()
    return true

func equip_item(slot_idx: int) -> bool:
    var item: ItemData = slots[slot_idx]
    if not item: return false

    var equip_slot = _get_equip_slot(item.item_type)
    if equip_slot == "": return false

    # Unequip cái cũ → về inventory
    if equipped[equip_slot]:
        unequip_item(equip_slot)

    # Equip mới
    equipped[equip_slot] = item
    slots[slot_idx] = null
    stats.apply_item_bonus(item, true)
    item_equipped.emit(equip_slot, item)
    EventBus.item_equipped.emit(item, equip_slot)
    inventory_changed.emit()
    return true

func unequip_item(equip_slot: String):
    var item = equipped[equip_slot]
    if not item: return
    equipped[equip_slot] = null
    stats.apply_item_bonus(item, false)
    add_item(item)

func _get_equip_slot(type: ItemData.ItemType) -> String:
    match type:
        ItemData.ItemType.WEAPON:  return "weapon"
        ItemData.ItemType.HELMET:  return "helmet"
        ItemData.ItemType.CHEST:   return "chest"
        ItemData.ItemType.BOOTS:   return "boots"
    return ""

# wave_manager.gd
extends Node

@export var enemy_scenes: Array[PackedScene] = []
@export var spawn_points: Array[Node2D]     = []

var current_wave: int = 0
var enemies_alive: int = 0

signal wave_started(wave_num)
signal wave_cleared(wave_num)
signal all_waves_done

func start_next_wave():
    current_wave += 1
    wave_started.emit(current_wave)

    # Số lượng và loại enemy tăng dần
    var count = 5 + current_wave * 3
    enemies_alive = count

    for i in count:
        await get_tree().create_timer(0.5).timeout
        _spawn_enemy()

func _spawn_enemy():
    var spawn = spawn_points[randi() % spawn_points.size()]
    # Chọn enemy type theo wave (khó dần)
    var tier = min(current_wave / 3, enemy_scenes.size() - 1)
    var enemy = enemy_scenes[tier].instantiate()
    add_child(enemy)
    enemy.global_position = spawn.global_position
    enemy.get_node("HealthComponent").died.connect(_on_enemy_died)

func _on_enemy_died():
    enemies_alive -= 1
    if enemies_alive <= 0:
        wave_cleared.emit(current_wave)
        await get_tree().create_timer(3.0).timeout
        start_next_wave()

func shake(str=6.0, dur=0.25):
  var t = 0.0
  while t < dur:
    offset = Vector2(randf_range(-str,str),
             randf_range(-str,str))
    t += await get_tree().process_frame
    t += get_process_delta_time()
  offset = Vector2.ZERO
